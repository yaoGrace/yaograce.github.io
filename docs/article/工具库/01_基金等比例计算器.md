        // 自定义提示框函数
        function showCustomAlert(message) {
            // 创建遮罩层
            const overlay = document.createElement('div');
            overlay.className = 'overlay';
            document.body.appendChild(overlay);

            // 创建提示框容器
            const alertBox = document.createElement('div');
            alertBox.className = 'custom-alert';
            
            // 提示框内容
            alertBox.innerHTML = `
                <div class="alert-header">
                    <span class="alert-icon"></span>
                    <div class="alert-title">温馨提示</div>
                </div>
                <div class="alert-body">${message}</div>
                <div class="alert-footer">
                    <button class="alert-btn" onclick="closeCustomAlert()">确 定</button>
                </div>
            `;
            
            // 添加到页面
            document.body.appendChild(alertBox);
            
            // 聚焦输入框（关闭提示后）
            const aCurrentInput = document.getElementById('aCurrent');
            alertBox.querySelector('.alert-btn').addEventListener('click', () => {
                aCurrentInput.focus();
            });
        }

        // 关闭自定义提示框
        function closeCustomAlert() {
            const overlay = document.querySelector('.overlay');
            const alertBox = document.querySelector('.custom-alert');
            if (overlay) overlay.remove();
            if (alertBox) alertBox.remove();
        }

        // 核心计算逻辑
        function calculateRatio() {
            // 获取输入元素
            const aTotalInput = document.getElementById('aTotal');
            const bTotalInput = document.getElementById('bTotal');
            const aCurrentInput = document.getElementById('aCurrent');
            
            // 获取输入值并去除首尾空格
            const aTotalStr = aTotalInput.value.trim();
            const bTotalStr = bTotalInput.value.trim();
            const aCurrentStr = aCurrentInput.value.trim();

            // 1. 校验是否为有效数字
            if (!/^\d+(\.\d+)?$/.test(aTotalStr)) {
                showCustomAlert('【天道轮回】最大投资金额请输入有效的数字（如 100000 或 100000.00）');
                aTotalInput.focus();
                return;
            }
            if (!/^\d+(\.\d+)?$/.test(bTotalStr)) {
                showCustomAlert('【我】最大投资金额请输入有效的数字（如 3000 或 3000.00）');
                bTotalInput.focus();
                return;
            }
            if (!/^\d+(\.\d+)?$/.test(aCurrentStr)) {
                showCustomAlert('【天道轮回】本次投资金额请输入有效的数字（如 5000 或 5000.00），不支持非数字字符哦～');
                aCurrentInput.focus();
                return;
            }

            // 转换为数值
            const aTotal = parseFloat(aTotalStr);
            const bTotal = parseFloat(bTotalStr);
            const aCurrent = parseFloat(aCurrentStr);

            // 2. 校验数值范围
            if (aTotal <= 0) {
                showCustomAlert('【天道轮回】每个基金最大投资金额不能为0或负数');
                aTotalInput.focus();
                return;
            }
            if (bTotal < 0) {
                showCustomAlert('【我】每个基金最大投资金额不能为负数');
                bTotalInput.focus();
                return;
            }
            if (aCurrent <= 0) {
                showCustomAlert('请输入【天道轮回】本次投资的有效金额（大于0），这是计算的必填项哦～');
                aCurrentInput.focus();
                return;
            }
            // 3. 新增：校验本次投资金额不超过最大投资金额
            if (aCurrent > aTotal) {
                showCustomAlert(`【天道轮回】本次投资金额不能超过其最大投资金额（${aTotal} 元），请重新输入～`);
                aCurrentInput.focus();
                // 选中输入框内容，方便用户直接修改
                aCurrentInput.select();
                return;
            }

            // 核心计算
            const ratio = bTotal / aTotal; // 基础比例
            const bCurrent = (aCurrent * ratio).toFixed(2); // 【我】应投金额（保留2位小数）
            const ratioPercent = (ratio * 100).toFixed(4); // 比例百分比（保留4位小数）

            // 完整展示计算结果
            document.getElementById('bCurrent').textContent = `${bCurrent} 元`;
            document.getElementById('ratio').textContent = ratio.toFixed(6);
            document.getElementById('ratioPercent').textContent = `${ratioPercent}%`;
            
            // 更新计算明细（展示具体数值）
            const detailText = `计算逻辑：${bCurrent} 元 = ${aCurrent} 元 × (${bTotal} 元 ÷ ${aTotal} 元) = ${aCurrent} 元 × ${ratio.toFixed(6)}`;
            document.getElementById('calculationDetail').textContent = detailText;

            // 显示结果区域并平滑滚动
            document.getElementById('resultArea').style.display = 'block';
            document.getElementById('resultArea').scrollIntoView({ 
                behavior: 'smooth',
                block: 'nearest'
            });
        }

        // 监听输入框回车事件
        document.getElementById('aCurrent').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') calculateRatio();
        });
        document.getElementById('bTotal').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') calculateRatio();
